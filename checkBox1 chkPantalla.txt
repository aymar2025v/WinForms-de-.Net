namespace PROYECTO_2
{
	public partial class Form1 : Form
	{

		public Form1()
		{
			InitializeComponent();
		}

		private void btnTotal_Click(object sender, EventArgs e)
		{
			int total = 0;
			if (chkPantalla.Checked == true)
			{
				total += 100;
			}
			if (chkTeclado.Checked == true)
			{
				total += 100;
			}
			if (chkMemoria.Checked == true)
			{
				total += 300;
			}

			//lblMensaje.Text = "Total: $" + total.ToString();
			MessageBox.Show("Total: $" + total.ToString());
		}

		private void chkPantalla_CheckedChanged(object sender, EventArgs e)
		{
			if (chkPantalla.Checked == true)
				lblPantalla.Text = "Pantalla $100 (Seleccionada)";
			else
				lblPantalla.Text = "";
		}

		private void chkTeclado_CheckedChanged(object sender, EventArgs e)
		{
			if (chkTeclado.Checked == true)
				lblTeclado.Text = "Teclado $50 (Seleccionado)";
			else
				lblTeclado.Text = "";
		}

		private void chkMemoria_CheckedChanged(object sender, EventArgs e)
		{
			if(chkMemoria.Checked == true)					
				lblMemoria.Text = "Memoria $300 (Seleccionada)";
			else
				lblMemoria.Text = "";
		}
	}
}
